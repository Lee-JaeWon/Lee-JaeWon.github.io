#include <iostream>

void dfs(int start, vector<vector<int>> v, bool check[]) {
    if (check[start] == true) {
        return;
    }
    check[start] = true;
    cout << start << " ";

    for (int i = 0; i < v[start].size(); i++) {
        int next = v[start][i];

        if (!(check[next]))
            dfs(next, v, check);
    }
}
void bfs(int start, vector<vector<int>> v, bool check[]) {
    queue<int> q;

    q.push(start);
    check[start] = true;

    while (!q.empty()) {
        int res = q.front();
        q.pop();
        cout << res << " ";

        for (int i = 0; i < v[res].size(); i++) {
            if (check[v[res][i]] == false) {
                q.push(v[res][i]);
                check[v[res][i]] = true;
            }
        }
    }
}
